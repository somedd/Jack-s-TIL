~~~~~~~~~~~~~javascript
var bynum = parseInt(prompt('x by x 수를 입력하세요.'));
var gapnum = parseInt(prompt('간격 수를 입력하세요.'));
str = '';
for (var i = 1; i <= bynum; i++) {
  for (var j = 1; j <= bynum * gapnum; j = j + gapnum) {
    if (i == 1) {
      str = str + ' ' + j;
    } else {
      str = str + ' ' + (gapnum * (i - 1) + j);
    }
  }
  str = str + '\n';
}
console.log(str);
~~~~~~~~~~~~~
